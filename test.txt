# Тест 1: Запит картинки якої ще немає (має повернути 404)
curl -i -X GET http://localhost:3000/200

# Тест 2: Запис картинки в кеш
# (Для цього тесту скачай будь-яку маленьку картинку 200.jpg у папку з проєктом)
curl -i -X PUT http://localhost:3000/200 --data-binary "@200.jpg"

# Тест 3: Отримання картинки з кешу (має повернути 200)
curl -i -X GET http://localhost:3000/200

# Тест 4: Видалення картинки з кешу (має повернути 200)
curl -i -X DELETE http://localhost:3000/200

# Тест 5: Недозволений метод (має повернути 405)
curl -i -X POST http://localhost:3000/200
